int xoa1(char a[], int q)
{
    int n = strlen(a), i;
    for (i = q; i <= n; i++)
        {
         a[i] = a[i + 1];
        }
    n--;
    return a;
}

int xoa(char a[])
{
    int i, n=strlen(a);
    for (i = 0; i < n; i++)
        {
            if (a[0] == 32)
                {
                    xoa1(a, 0);
                    n--;
                }
            if (a[n-1] == 32)
                {
                    xoa1(a, n-1);
                    n--;
                }
            if (a[i] == 32 && a[i + 1] == 32)
                {
                    xoa1(a, i);
                    i--;
                    n--;
                }
    }
    return a;
}

char inhoa(char s[])
{
  int i;
if(s[0]>='a'&&s[0]<='z')                              
      {
         s[0]=s[0]-32;
         }
     for(i=1;i<strlen(s);i++)
        {
if(s[i]==' ' && s[i+1]>='a' && s[i+1]<='z' )
         {
     s[i+1]=s[i+1]-32;
           }        
               }
return s;
}
